let cart = [];

function renderProducts(filter="all"){
  const grid = document.getElementById('productGrid');
  grid.innerHTML = "";
  products.filter(p => filter==="all" || p.cat===filter).forEach(p=>{
    grid.innerHTML += `
      <div class="product">
        <div class="img">${p.icon}</div>
        <div class="info">
          <div class="tag">${p.tag}</div>
          <h4>${p.name}</h4>
          <div class="price">$${p.price.toFixed(2)}</div>
          <button onclick="addToCart(${p.id})">🛒 Add to Cart</button>
        </div>
      </div>`;
  });

function filterCat(cat, btn){
  renderProducts(cat);
  document.querySelectorAll('.filters button').forEach(b=>b.classList.remove('active'));
  if(btn) btn.classList.add('active');
  document.getElementById('shop').scrollIntoView({behavior:'smooth'});
}

function addToCart(id){
  const p = products.find(x=>x.id===id);
  const exist = cart.find(x=>x.id===id);
  if(exist){ exist.qty++; } else { cart.push({...p, qty:1}); }
  updateCart();
  showToast(p.name + " added!");
}

function removeFromCart(id){ cart = cart.filter(x=>x.id!==id); updateCart(); }

function updateCart(){
  document.getElementById('cartCount').innerText = cart.reduce((s,i)=>s+i.qty,0);
  const box = document.getElementById('cartItems');
  if(cart.length===0){ box.innerHTML = '<div class="empty">🛒 Your cart is empty</div>'; }
  else {
    box.innerHTML = cart.map(i=>`
      <div class="cart-item">
        <div>${i.icon} ${i.name}<br><small>$${i.price.toFixed(2)} × ${i.qty}</small></div>
        <div style="text-align:right;">$${(i.price*i.qty).toFixed(2)}<br>
        <span class="rm" onclick="removeFromCart(${i.id})">Remove</span></div>
      </div>`).join('');
  }
  document.getElementById('cartTotal').innerText =
    cart.reduce((s,i)=>s+i.price*i.qty,0).toFixed(2);
}

function openCart(){ document.getElementById('drawer').classList.add('open'); document.getElementById('overlay').classList.add('open'); }
function closeCart(){ document.getElementById('drawer').classList.remove('open'); document.getElementById('overlay').classList.remove('open'); }

function checkout(){
  if(cart.length===0){ showToast("Cart is empty!"); return; }

}
